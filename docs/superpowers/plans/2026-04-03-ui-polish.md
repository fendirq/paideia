### Task 5: Subject Chips Component

**Files:**
- Create: `src/components/subject-chips.tsx`

- [ ] **Step 1: Create reusable subject chips component**

```tsx
// src/components/subject-chips.tsx
"use client";

const SUBJECT_OPTIONS = [
  { value: "MATHEMATICS", label: "Math", color: "#5b9bd5" },
  { value: "ENGLISH", label: "English", color: "#c57bdb" },
  { value: "HISTORY", label: "History", color: "#e8a838" },
  { value: "SCIENCE", label: "Science", color: "#4a9d5b" },
  { value: "HUMANITIES", label: "Humanities", color: "#d4a574" },
  { value: "MANDARIN", label: "Mandarin", color: "#e87838" },
  { value: "OTHER", label: "Other", color: "#a39e98" },
];

interface SubjectChipsProps {
  value: string;
  onChange: (value: string) => void;
}

export function SubjectChips({ value, onChange }: SubjectChipsProps) {
  return (
    <div className="flex flex-wrap gap-2">
      {SUBJECT_OPTIONS.map((opt) => (
        <button
          key={opt.value}
          type="button"
          onClick={() => onChange(opt.value)}
          className={`flex items-center gap-[7px] px-4 py-[9px] rounded-[20px] font-display text-[13px] font-medium transition-all border ${
            value === opt.value
              ? "border-accent bg-accent/[0.08] text-text-primary"
              : "border-white/[0.05] bg-bg-surface text-text-secondary hover:border-white/[0.1]"
          }`}
        >
          <span
            className="w-[7px] h-[7px] rounded-full"
            style={{ background: opt.color }}
          />
          {opt.label}
        </button>
      ))}
    </div>
  );
}
```

```tsx
// src/components/activity-chart.tsx
"use client";

interface ActivityChartProps {
  data: { day: string; count: number }[];
}

export function ActivityChart({ data }: ActivityChartProps) {
  const max = Math.max(...data.map((d) => d.count), 1);
  return (
    <div className="bg-bg-inner border border-white/[0.04] rounded-[14px] p-6">
      <h3 className="font-display text-[13px] font-semibold mb-4">Weekly Activity</h3>
      <div className="flex items-end gap-2 h-[120px]">
        {data.map((d) => (
          <div key={d.day} className="flex-1 flex flex-col items-center gap-1.5">
            <div
              className="w-full rounded-t bg-accent/70 hover:bg-accent transition-colors"
              style={{ height: `${(d.count / max) * 100}%`, minHeight: d.count > 0 ? "4px" : "0" }}
            />
            <span className="text-[10px] text-text-muted font-display">{d.day}</span>
          </div>
        ))}
      </div>
    </div>
  );
}
```
